sain.vec

A contiguous growable alternative to builtin list with extra functionalities.

When initializing a vec, it will not build the underlying list until the first element gets pushed. Which saves a little bit of memory.

Example
names = Vec()

names.push('foo')
names.push('bar')

print(names) # ['foo', 'bar']
assert names.len() == 2

Equality is supported.

vec = Vec((1,2,3))
vec == [1, 2, 3] and vec # True

214    def split_off(self, at: int) -> Vec[T]:
215        """Split the vector off at the specified position.
216
217        Example
218        -------
219        ```py
220        vec = Vec((1, 2, 3))
221        split = vec.split_off(1)
222
223        print(split, vec)  # [1], [2, 3]
224        ```
225        """
226
227        assert self._ptr is not None, "Can't access an empty sequence."
228        if at > self.len():
229            raise RuntimeError(
230                f"Index `at` ({at}) should be <= than len of vector ({self.len()}) "
231            ) from None
232
233        # No reason to split at 0.
234        if at == 0:
235            return self
236
237        split = self[at : self.len()]
238        self._ptr = self._ptr[0:at]
239        return split

241    def split_first(self) -> _option.Option[tuple[T, collections.Sequence[T]]]:
242        """Split the first and rest elements of the vector,

397    def swap_remove(self, item: T) -> T:
398        """Remove the first appearance of `item` from this vector and return it.
399
400        This will raise a `ValueError` if `item` is not in this vector.
401
402        Example
403        -------
404        ```py
405        vec = Vector(('a', 'b', 'c'))
406        vec.remove('a')
407        assert vec == ['b', 'c']
408        ```
409        """
410        assert self._ptr is not None, "Can't access an empty sequence."
411        try:
412            i = next(i for i in self._ptr if i == item)
413        except StopIteration:
414            raise ValueError(f"Item `{item}` not in list") from None
415
416        self.remove(i)
417        return i
