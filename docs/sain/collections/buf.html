 73ENCODING = "utf-8"
 74REPLACEMENT_CHARACTER = "�"

 77def unwrap_bytes(data: Rawish) -> bytes:
 78    if isinstance(data, _io.StringIO):
 79        buf = bytes(data.read(), encoding=ENCODING)
 80    elif isinstance(data, memoryview):
 81        buf = data.tobytes()
 82    else:
 83        # BufferedReader | BytesIO
 84        buf = data.read()
 85    return buf

 88@typing.final
 89class Bytes:
 90    """Provides abstractions for working with UTF-8 compatible bytes.
 91
 92    It is an efficient container for storing and operating with bytes.
 93    It behaves very much like `array.array[int]` as well has the same layout.
 94
 95    A `Bytes` objects are usually used within networking applications, but can also be used
 96    elsewhere as well.
 97
 98    Construction
 99    ------------
100    `Bytes` object accept multiple rawish data types, See `Rawish` for all supported types.
101
102    * `Bytes()`: Initialize an empty `Bytes` object
103    * `from_str`: Create `Bytes` from `str`
104    * `from_bytes`: Create `Bytes` from a `Buffer` type
105    * `from_raw`: Create `Bytes` from a `Rawish` type
106    * `from_static`: Create `Bytes` that points to an `array.array[int]` without copying it
107
108    Example
109    -------
110    ```py
111    from sain import Bytes
112
113    buf = Bytes()
114    buffer.put_bytes(b"Hello")
115    print(buffer) # [72, 101, 108, 108, 111]
116
117    buf.put(1234)
118    assert buffer.as_bytes() == b"Hello\x04\xd2"
119    ```
120    """

182        """Initialize a new `Bytes` in-place from a valid raw data type.

207    def try_to_str(self) -> Result[str, bytes]:
208        """A safe method to convert `self` into a string.
209
210        This may fail if the `self` contains invalid bytes. strings
211        needs to be valid utf-8.
212
213        Example
214        -------
215        ```py
216        buf = Bytes()
217        sparkles_heart = [240, 159, 146, 150]
218        buf.put(sparkles_heart)
219
220        assert buf.try_to_str().unwrap() == "💖"
221        ```
222
223        Incorrect bytes
224        ---------------
225        ```py
226        invalid_bytes = Bytes.from_raw([0, 159, 146, 150])
227        invalid_bytes.try_to_str().is_err()
228        ```
229
230        Returns
231        -------
232        `Result[str, bytes]`
233            If successful, returns the decoded string, otherwise the original bytes that failed
234            to get decoded.
235        """
236        try:
237            return _result.Ok(self.as_bytes().decode(ENCODING))
238        except UnicodeDecodeError as e:
239            return _result.Err(e.object)

241    def to_str(self) -> str:
242        """Convert `self` to a utf-8 string. which may or may not be expensive.
243
244        During the conversion process, any invalid bytes will get converted to
245        [REPLACEMENT_CHARACTER](https://en.wikipedia.org/wiki/Specials_(Unicode_block))
246        which looks like this `�`, so be carful on what you're trying to convert.
247
248        Use `.try_to_str` try attempt the conversion incase of failure.
249
250        Example
251        -------
252        ```py
253        buff = Bytes()
254        sparkles_heart = [240, 159, 146, 150]
255        buff.put(sparkles_heart)
256
257        assert buf.to_str() == "💖"
258        ```
259
260        Incorrect bytes
261        ---------------
262        ```py
263        invalid_bytes = Bytes.from_raw(b"Hello \xf0\x90\x80World")
264        assert invalid_bytes.to_str() == "Hello �World"
265        ```
266
267        Unrecoverable failures
268        ----------------------
269        This method may raise `UnicodeDecodeError` if `self` contains
270        invalid bytes that cannot be decoded.
271
272        ```py
273        invalid_bytes = Bytes.from_raw([0, 159, 146, 150])
274        invalid_bytes.to_str() # ERROR
275        ```
276        """
277        if not self._buf:
278            return ""
279
280        bytes_ = self._buf.tobytes()
281        try:
282            return bytes_.decode(ENCODING)
283        except UnicodeDecodeError as err:
284            start, end = err.start, err.end
285            # we need to create a new copy of the bytes to replace the invalid bytes.
286            new = bytes_.replace(bytes_[start:end], REPLACEMENT_CHARACTER.encode())
287            return new.decode(ENCODING)

291    def leak(self) -> Option[array.array[int]]:
292        """Consumes and leaks the `Bytes`, returning the underlying `array` and setting it to `None`.
293
294        `None` is returned if it was already `None`.
295
296        Example
297        -------
298        ```py
299        bytes = Bytes.from_str("chunks of data")
300        consumed = bytes.leak().unwrap()
301
302        assert bytes.is_empty()
303        assert consumed.tobytes() == b"chunks of data"
304        ```
305        """
306        if self._buf is None:
307            return _option.nothing_unchecked()
308
309        arr = self._buf
310        self._buf = None
311        return _option.Some(arr)

367    def put(self, src: int) -> None:
368        """Append a byte at the end of the array.
369
370        unlike `.put_bytes`, this method appends instead of extending the array
371        which is faster if you're putting a single byte in a single call.
372
373        if `self` hasn't been initialized, the array will allocate along with the byte.
374
375        Example
376        -------
377        ```py
378        buf = Bytes()
379        buf.put(32)
380        assert buf.as_bytes() == b' '
381        ```
382
383        Parameters
384        ----------
385        src : `int`
386            An unsigned integer, also known as `u8` to put.
387
388        Raises
389        ------
390        `OverflowError`
391            If a byte in `src` not in range 0..=255
392        """
393        if self._buf is None:
394            # If it was `None`, we initialize it with a source instead of appending.
395            self._buf = array.array("B", (src,))
396        else:
397            self._buf.append(src)

399    def put_raw(self, src: Rawish) -> None:
400        """Extend `self` from a raw data type source.
401
402        Example
403        -------
404        ```py
405        buffer = Bytes()
406        # A file descriptor's contents
407        with open('file.txt', 'rb') as file:
408            buffer.put_raw(file)
409
410        # bytes io
411        buffer.put(io.BytesIO(b"data"))
412        # string io
413        buffer.put(io.StringIO("data"))
414        ```
415
416        Parameters
417        ----------
418        src : `Rawish`
419            A valid raw data type. See `Rawish` for more details.
420
421        Raises
422        ------
423        `OverflowError`
424            If a byte in `src` is greater than `u8->MAX` which is 255.
425        """
426        if self._buf is None:
427            # If it was `None`, we initialize it with a source instead of extending.
428            self._buf = array.array("B", unwrap_bytes(src))
429        else:
430            self._buf.extend(unwrap_bytes(src))

443        Parameters

244
245        During the conversion process, any invalid bytes will get converted to
246        [REPLACEMENT_CHARACTER](https://en.wikipedia.org/wiki/Specials_(Unicode_block))

285            start, end = err.start, err.end

Provides abstractions for working with UTF-8 compatible bytes.

It is an efficient container for storing and operating with bytes. It behaves very much like array.array[int] as well has the same layout.

A Bytes objects are usually used within networking applications, but can also be used elsewhere as well.

Construction

Bytes object accept multiple rawish data types, See Rawish for all supported types.

  • Bytes(): Initialize an empty Bytes object
  • from_str: Create Bytes from str
  • from_bytes: Create Bytes from a Buffer type
  • from_raw: Create Bytes from a Rawish type
  • from_static: Create Bytes that points to an array.array[int] without copying it
Example
from sain import Bytes

buf = Bytes()
buffer.put_bytes(b"Hello")
print(buffer) # [72, 101, 108, 108, 111]

buf.put(1234)
assert buffer.as_bytes() == b"HelloÒ"
