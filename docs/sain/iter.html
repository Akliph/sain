  98class Iterator(
  99    typing.Generic[Item],
 100    abc.ABC,
 101    _default.Default["Empty[Item]"],
 102):
 103    """An abstract interface for dealing with iterators.
 104
 105    This is exactly the same trait as `core::iter::Iterator` trait from Rust.
 106
 107    This is the main interface that any type can implement by basically inheriting from it.
 108    The method `__next__` is the only method that needs to be implemented, You get all the other methods for free.
 109
 110    If you want to use a ready iterator for general purposes, Use `Iter`. This interface is only for implementors
 111    and type hints.
 112
 113    Example
 114    -------
 115    ```py
 116    @dataclass
 117    class Message:
 118        content: str
 119
 120    class MessageIterator(sain.Iterator[Message]):
 121        def __init__(self, id: int = random.randint(0, 100)) -> None:
 122            self._session: requests.Session | None = None
 123            self.id = id
 124
 125        def __next__(self) -> Message:
 126            if self._session is None:
 127                self._session = requests.session()
 128
 129            try:
 130                with self._session:
 131                    response = self._session.get(f"https://dummyjson.com/products/{self.id}").json()
 132            finally:
 133                self._session = None
 134
 135            return Message(response["description"])
 136
 137    it = MessageIterator()
 138    # Lazily fetch the first 5 messages from the API.
 139    for msg in it.take(5):
 140        print(msg)
 141    ```
 142    """

 176    @typing.final
 177    def collect(
 178        self, *, cast: collections.Callable[[Item], OtherItem] | None = None
 179    ) -> collections.Sequence[Item] | collections.Sequence[OtherItem]:
 180        """Collects all items in the iterator into an immutable sequence.
 181
 182        Example
 183        -------
 184        ```py
 185        iterator = Iter(range(3))
 186        iterator.collect()
 187        # (0, 1, 2, 3)
 188        iterator.collect(cast=str) # Map each element and collect it.
 189        # ('0', '1', '2', '3')
 190        ```
 191
 192        Parameters
 193        ----------
 194        cast: `T | None`
 195            An optional type to cast the items into.
 196            If not provided the items will be returned as it's original type.
 197        """
 198        if cast is not None:
 199            return tuple(cast(i) for i in self)
 200
 201        return tuple(_ for _ in self)

 203    @typing.final
 204    def collect_into(self, collection: Collector[Item]) -> None:
 205        """Consume this iterator, extending all items in the iterator into a mutable `collection`.
 206
 207        Example
 208        -------
 209        ```py
 210        iterator = Iter([1, 1, 2, 3, 4, 2, 6])
 211        uniques = set()
 212        iterator.collect_into(uniques)
 213        # (1, 2, 3, 4, 6)
 214        ```
 215
 216        Parameters
 217        ----------
 218        collection: `MutableSequence[T]` | `set[T]`
 219            The collection to extend the items in this iterator with.
 220        """
 221        if isinstance(collection, collections.MutableSequence):
 222            collection.extend(_ for _ in self)
 223        else:
 224            collection.update(_ for _ in self)

 256    @typing.final
 257    def raw_parts(self) -> collections.Generator[Item, None, None]:
 258        """Decompose all elements from this iterator, yielding it one by one
 259        as a normal generator.
 260
 261        This mainly used for objects that needs to satisfy its exact type.
 262
 263        ```py
 264        it = Iter("cba")
 265        sort = sorted(it.raw_parts())
 266
 267        assert it.count() == 0
 268        assert sort == ["a", "b", "c"]
 269        ```
 270        """
 271        for item in self:
 272            yield item

 278    def next(self) -> Option[Item]:
 279        """Advance the iterator, Returning the next item, `Some(None)` if all items yielded.
 280
 281        Example
 282        -------
 283        ```py
 284        iterator = Iter(["1", "2"])
 285        assert iterator.next() == Some("1")
 286        assert iterator.next() == Some("2")
 287        assert iterator.next().is_none()
 288        ```
 289        """
 290        try:
 291            return _option.Some(self.__next__())
 292        except StopIteration:
 293            # ! SAFETY: No more items in the iterator.
 294            return _option.NOTHING  # pyright: ignore

 296    def cloned(self) -> Cloned[Item]:
 297        """Creates an iterator which shallow copies its elements by reference.
 298
 299        If you need a copy of the actual iterator and not the elements.
 300        use `Iter.clone()`
 301
 302        .. note::
 303            This method calls [`copy.copy()`](https://docs.python.org/3/library/copy.html)
 304            on each item that is being yielded.
 305
 306        Example
 307        -------
 308        ```py
 309        @dataclass
 310        class User:
 311            users_ids: list[int] = []
 312
 313        # An iterator which elements points to the same user.
 314        user = User()
 315        it = Iter((user, user))
 316
 317        for u in it.cloned():
 318            u.user_ids.append(1)
 319
 320        # We iterated over the same user pointer twice and appended "1"
 321        # since `copy` returns a shallow copy of nested structures.
 322        assert len(user.user_ids) == 2
 323        ```
 324        """
 325        return Cloned(self)
 326
 327    def copied(self) -> Copied[Item]:
 328        """Creates an iterator which copies all of its elements by value.
 329
 330        If you only need a copy of the item reference, Use `.cloned()` instead.
 331
 332        .. note::
 333            This method simply calls [`copy.deepcopy()`](https://docs.python.org/3/library/copy.html)
 334            on each item that is being yielded.
 335
 336        Example
 337        -------
 338        ```py
 339        @dataclass
 340        class User:
 341            users_ids: list[int] = []
 342
 343        # An iterator which elements points to the same user.
 344        user = User()
 345        it = Iter((user, user))
 346
 347        for u in it.copied():
 348            # A new list is created for each item.
 349            u.user_ids.append(1)
 350
 351        # The actual list is untouched since we consumed a deep copy of it.
 352        assert len(user.user_ids) == 0
 353        ```
 354        """
 355        return Copied(self)

 667    def find(self, predicate: collections.Callable[[Item], bool]) -> Option[Item]:
 668        """Searches for an element of an iterator that satisfies a predicate.
 669
 670        `find()` takes a closure that returns true or false. It applies this closure to each element of the iterator,
 671        and if any of them return true,
 672        then find() returns `Some(element)`. If they all return false, it returns None.
 673
 674        Example
 675        -------
 676        ```py
 677        it = Iter(range(10))
 678        item = it.find(lambda num: num > 5)
 679        print(item) # 6
 680        ```
 681        """
 682        while True:
 683            try:
 684                n = self.__next__()
 685                if predicate(n):
 686                    return _option.Some(n)
 687            except StopIteration:
 688                # ! SAFETY: No more items to search.
 689                # we also want to avoid function calls.
 690                return _option.NOTHING  # pyright: ignore

 766@typing.final
 767@diagnostic
 768class Iter(Iterator[Item]):
 769    """a lazy iterator that has its items ready in-memory.
 770
 771    This is similar to Rust `std::slice::Iter<T>` item which iterables can build
 772    from this via `.iter()` method.
 773
 774    Example
 775    -------
 776    ```py
 777    iterator = Iter([1, 2, 3])
 778
 779    # Limit the results to 2.
 780    for item in iterator.take(2):
 781        print(item)
 782    # 1
 783    # 2
 784
 785    # Filter the results.
 786    for item in iterator.filter(lambda item: item > 1):
 787        print(item)
 788    # 2
 789    # 3
 790    # 3

An abstract interface for dealing with iterators.

This is exactly the same trait as core::iter::Iterator trait from Rust.

This is the main interface that any type can implement by basically inheriting from it. The method __next__ is the only method that needs to be implemented, You get all the other methods for free.

If you want to use a ready iterator for general purposes, Use Iter. This interface is only for implementors and type hints.

Example
@dataclass

    content: str

class MessageIterator(Iterator[Message]):
    def __init__(self, id: int = random.randint(0, 100)) -> None:
        self._session: requests.Session | None = None
        self.id = id

            self._session = requests.session()

        try:
            with self._session:
                response = self._session.get(f"https://dummyjson.com/products/{self.id}").json()

        return Message(response["description"])
