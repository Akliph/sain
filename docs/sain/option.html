 65@typing.final
 66class Some(typing.Generic[ValueT], _default.Default[None]):
 67    """The `Some` object. A typesafe object is used for values that might be `T` or `None`.
 68
 69    It is similar to `typing.Optional[T]`, But has proper methods to handle the contined value.
 70
 71    Example
 72    -------
 73    ```py
 74    value = Some("Hello")
 75    print(value)
 76    # Some("Hello")
 77
 78    # This will unwrap the contained value as long as
 79    # it is not `None` otherwise this will raise an error.
 80    print(value.unwrap())
 81    # "Hello"
 82
 83    none_value = Some(None)
 84    while none_value.unwrap():
 85        # Never unreachable!
 86
 87    # Solving it with `unwrap_or` method to unwrap the value or return a default value.
 88    print(none_value.unwrap_or(10))
 89    # 10
 90    ```
 91    """
 92
 93    __slots__ = ("_value",)
 94
 95    def __init__(self, value: ValueT | None, /) -> None:
 96        self._value = value
 97
 98    @staticmethod
 99    def default() -> None:
100        """Return the default value for `Some`. This always returns `None`."""
101        return None
102
103    @property
104    def read(self) -> ValueT | None:
105        """Read the contained value."""
106        return self._value
107
108    def unwrap(self) -> ValueT:
109        """Unwrap the inner value either returning if its not `None` or raising a `RuntimeError`.
110
111        Example
112        -------
113        ```py
114        value = Some(5)
115        print(value.unwrap())
116        # 5
117
118        value = Some(None)
119        print(value.unwrap())
120        # RuntimeError
121        ```
122
123        Raises
124        ------
125        `RuntimeError`
126            If the inner value is `None`.
127        """
128        if self._value is None:
129            raise RuntimeError(f"Can't unwrap {type(self).__name__} value.") from None
130
131        return self._value
132
133    def unwrap_or(self, default: ValueT, /) -> ValueT:
134        """Unwrap the inner value either returning if its not `None` or returning `default`.
135
136        Example
137        -------
138        ```py
139        value = Some(5)
140        print(value.unwrap_or(10))
141        # 5
142
143        # Type hint is required here.
144
145        value: Some[int] = Some(None)
146        print(value.unwrap_or(10))
147        # 10
148        ```
149        """
150        if self._value is None:
151            return default
152
153        return self._value
154
155    def unwrap_or_else(self, f: FnOnce[ValueT], /) -> ValueT:
156        """Unwrap the inner value either returning if its not `None` or calling `f` to get a default value.
157
158        Example
159        -------
160        ```py
161        value = Some(5)
162        print(value.unwrap_or_else(lambda: 10))
163        # 5
164
165        value: Some[bool] = Some(None)
166        print(value.unwrap_or_else(lambda: True))
167        # True
168        ```
169        """
170        if self._value is None:
171            return f()
172
173        return self._value
174
175    # Functional.
176    def map(self, f: Fn[ValueT, T], /) -> Some[T | None]:
177        """Map the inner value to a new value. Returning `Some[None]` if `ValueT` is `None`.
178
179        Example
180        -------
181        ```py
182        value = Some(5.0)
183
184        print(value.map(lambda x: x * 2.0))
185        # Some(10.0)
186
187        value: Some[bool] = Some(None)
188        print(value)
189        # Some(None)
190        ```
191        """
192        if self._value is None:
193            return Some(None)
194
195        return Some(f(self._value))
196
197    def map_or(self, default: T, f: Fn[ValueT, T], /) -> T:
198        """Map the inner value to a new value or return `default` if its `None`.
199
200        Example
201        -------
202        ```py
203        value = Some(5.0)
204
205        # Since the value is not `None` this will get mapped.
206        print(value.map_or(10.0, lambda x: x + 1.0))
207        # 6.0
208
209        # This is `None`, so the default value will be returned.
210        value: Some[str] = Some(None)
211        print(value.map_or("5", lambda x: str(x)))
212        # "5"
213        ```
214        """
215        if self._value is None:
216            return default
217
218        return f(self._value)
219
220    def map_or_else(self, default: FnOnce[T], f: typing.Callable[[ValueT], T], /) -> T:
221        """Map the inner value to a new value, Or return default which maps to `default()` if its `None`.
222
223        Example
224        -------
225        ```py
226        value = Some(5.0)
227
228        # Since the value is not `None` this will get mapped.
229        print(value.map_or_else(lambda: 10.0, lambda x: x + 1.0))
230        # 6.0
231
232        # This is `None`, so the default func will be returned.
233        value: Option[str] = Some(None)
234        print(value.map_or_else(lambda: "5", lambda x: str(x)))
235        # "5"
236        ```
237        """
238        if self._value is None:
239            return default()
240
241        return f(self._value)
242
243    def filter(self, predicate: Fn[ValueT, bool]) -> Some[ValueT]:
244        """Returns `Some[None]` if the contained value is `None`,
245
246        otherwise calls the predicate and returns `Some[ValueT]` if the predicate returns `True`.
247
248        Example
249        -------
250        ```py
251        value = Some([1, 2, 3])
252
253        print(value.filter(lambda x: len(x) >= 2))
254        # Some([1, 2, 3])
255
256        value: Some[int] = Some(None)
257        print(value.filter(lambda x: x > 3))
258        # None
259        ```
260        """
261        if (value := self._value) is not None:
262            if predicate(value):
263                return Some(value)
264
265        # We could return `None` here, but we want to return `Some[None]` instead.
266        return Some(None)
267
268    def take(self) -> None:
269        """Take the value from the `Some` object setting it to `None`.
270
271        Example
272        -------
273        ```py
274        value = Some("Hi")
275        value = value.take()
276        print(value)
277        # None
278        ```
279        """
280        v = self._value = None
281        return v
282
283    def replace(self, value: ValueT) -> Some[ValueT]:
284        """Replace the contained value with another value.
285
286        Example
287        -------
288        ```py
289        value = Some("Hi")
290        value = value.replace("Hello")
291
292        print(value)
293        # Some("Hello")
294        ```
295        """
296        v = self._value = value
297        return Some(v)
298
299    # Boolean checkings.
300    def expect(self, message: str, /) -> ValueT:
301        """Returns `ValueT` if the contained value is not `None` otherwise raises a `RuntimeError`.
302
303        Example
304        -------
305        ```py
306        value = Some("Hello")
307
308        print(value.expect("Value is None"))
309        # "Hello"
310
311        value: Some[str] = Some(None)
312        print(value.expect("Value is None"))
313        # RuntimeError("Value is None")
314        ```
315        """
316        if self._value is None:
317            raise RuntimeError(message)
318
319        return self._value
320
321    def and_ok(self, optb: Some[T | None]) -> Some[T | None]:
322        """Returns `Some[None]` if the contained value is `None`,
323
324        Otherwise return optb as `Some[T | None]` if optb is `Some[T]`.
325
326        Example
327        -------
328        ```py
329        value = Some(5)
330
331        print(value.and_ok(Some(10)))
332        # Some(10)
333
334        value: Some[int] = Some(10)
335        print(value.and_ok(Some(None)))  # optb is `None`.
336        # Some(None)
337        ```
338        """
339        if self._value is None:
340            return Some(None)
341
342        return optb
343
344    def and_then(self, f: Fn[ValueT, Some[T | None]]) -> Some[T | None]:
345        """Returns `Some[None]` if the contained value is `None`,
346
347        otherwise call `f` on `ValueT` and return `Some[T]` if it's value not `None`.
348
349        Example
350        -------
351        ```py
352        value = Some(5)
353        print(value.and_then(lambda x: Some(x * 2)))
354        # Some(10)
355
356        value: Some[int] = Some(10)
357        print(value.and_then(lambda x: Some(None)))
358        # Some(None)
359        ```
360        """
361        if self._value is None:
362            return Some(None)
363
364        return f(self._value)
365
366    def as_ref(self) -> Some[_ref.Ref[ValueT]]:
367        """Returns `Some[Ref[ValueT]]` if the contained value is not `None`, otherwise returns `None`.
368
369        Example
370        -------
371        ```py
372        value = Some(5).as_ref().unwrap()
373        owned = value.object
374        print(owned) # 5
375
376        # Create a copy of object.
377        clone = value.clone()
378        clone = 0
379        print(clone == owned) # False
380
381        value: Some[int] = Some(None)
382        print(value.as_ref())
383        # Some(Ref(None))
384        ```
385        """
386        if self._value is not None:
387            return Some(_ref.Ref(self._value))
388
389        return Some(None)
390
391    def is_some(self) -> bool:
392        """Returns `True` if the contained value is not `None`, otherwise returns `False`.
393
394        Example
395        -------
396        ```py
397        value = Some(5)
398        print(value.is_some())
399        # True
400
401        value: Some[int] = Some(None)
402        print(value.is_some())
403        # False
404        ```
405        """
406        return self._value is not None
407
408    def is_some_and(self, predicate: Fn[ValueT, bool]) -> bool:
409        """Returns `True` if the contained value is not `None` and
410        the predicate returns `True`, otherwise returns `False`.
411
412        Example
413        -------
414        ```py
415        value = Some(5)
416        print(value.is_some_and(lambda x: x > 3))
417        # True
418
419        value: Some[int] = Some(None)
420        print(value.is_some_and(lambda x: x > 3))
421        # False
422        ```
423        """
424        return self._value is not None and predicate(self._value)
425
426    def is_none(self) -> bool:
427        """Returns `True` if the contained value is `None`, otherwise returns `False`.
428
429        Example
430        -------
431        ```py
432        value = Some(5)
433        print(value.is_none())
434        # False
435
436        value: Some[int] = Some(None)
437        print(value.is_none())
438        # True
439        ```
440        """
441        return not self.is_some()
442
443
444    def __str__(self) -> str:
445        return f"Some({self._value!r})"
446
447    def __repr__(self) -> str:
448        return f"Some({self._value!r})"
449
450    def __bool__(self) -> bool:
451        return self.is_some()
452
453    def __eq__(self, other: object) -> bool:
454        if not isinstance(other, Some):
455            return NotImplemented
456
457        return self._value == other.read
458
459    def __ne__(self, other: object) -> bool:
460        return not self.__eq__(other)
461
462    def __hash__(self) -> int:
463        return hash(self._value)
